# Iniciar el agente si te da problemas y añadir la clave ssh
eval "$(ssh-agent -s)"			#Abre el agente
ssh-add ~/.ssh/ssh_key			#Agrega la key
ssh-add -l				#Comprueba que el ssh_key se agregó correctamente

# Añadir una clave SSH
cd ~/.ssh
ssh-keygen -t rsa -b 4096 -C "correo@gmail.com" #Genera la Key
eval "$(ssh-agent -s)"				#Abre el agente
ssh-add ~/.ssh/ssh_key				#Busca la ruta donde esté el archivo y el .pub
cat ~/.ssh/ssh_key.pub				#Agrega esta linea en github

# Crear un repositorio
mkdir proyecto
cd proyecto
git init				#Empiezas tu proyecto, en este caso agregar un script.sh
touch README.md
nano script.sh
chmod +x script.sh

# Publicar el proyecto
git add .
git commit -m "Comienzo del proyecto"	#Comienzo del proyecto
git branch -M master
#Te conectas a tu git poniendo el username/elrepositorio 
#creado anteriormente en git a través de https://github.com/ClickHerexe?tab=repositories
git remote add origin git@github.com:TU_USUARIO/TU_REPOSITORIO.git
git push -u origin master

# Actualizar un repositorio
git pull origin master
git add .
git commit -m "Descripción de los cambios"
git push origin master

# Descargar un repositorio
git clone git@github.com:TU_USUARIO/TU_REPOSITORIO.git
